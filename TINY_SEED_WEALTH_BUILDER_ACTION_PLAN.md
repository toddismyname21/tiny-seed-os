```javascript
  driftThreshold: 0.05,           // 5% absolute drift triggers rebalance
  useContributionsFirst: true,    // Tax-efficient: buy underweight first
  minTradeAmount: 10,             // Minimum trade size
  sellsFirst: true                // Execute sells before buys
};

function calculateDrift(current, target) {
  const drift = {};
  let totalDrift = 0;

  for (const [symbol, targetWeight] of Object.entries(target)) {
    const currentWeight = current[symbol] || 0;
    drift[symbol] = currentWeight - targetWeight;
    totalDrift += Math.abs(drift[symbol]);
  }

  return { drift, portfolioDrift: totalDrift / 2 };
}

function shouldRebalance(portfolioDrift) {
  return portfolioDrift >= REBALANCING_CONFIG.driftThreshold;
}
```

```javascript
function calculateSEPContribution(netSelfEmploymentIncome, employeeCount) {
  // Net SE income after SE tax deduction
  const seDeduction = netSelfEmploymentIncome * 0.5 * 0.153;
  const adjustedIncome = netSelfEmploymentIncome - seDeduction;

  // Maximum allowed
  const maxContribution = Math.min(
    70000,                          // 2025 limit
    adjustedIncome * 0.25          // 25% of adjusted income
  );

  // If employees exist, apply same percentage to all
  if (employeeCount > 0) {

      maxPerPerson: maxContribution,
      note: 'Same percentage must apply to all eligible employees'
    };
```
